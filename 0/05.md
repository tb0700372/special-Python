```Python
# 这是多线程示例
import threading
import time

b_time = time.time()

def a():
    print('a begin')
    time.sleep(1)
    print('a end')
def b():
    print('b begin')
    time.sleep(1)
    print('b end')

_a = threading.Thread(target=a)
_b = threading.Thread(target=b)

_a.start()
_b.start()

_a.join()
_b.join()

print(time.time() - b_time)  # 完成时间
```
